%%% Vertex styles

% Courtesy of
% https://tex.stackexchange.com/questions/59354/no-text-none-in-tikz
\makeatletter
\newif\iftikz@node@phantom
\tikzset{
  phantom/.is if=tikz@node@phantom,
  text/.code=%
    \edef\tikz@temp{#1}%
    \ifx\tikz@temp\tikz@nonetext
      \tikz@node@phantomtrue
    \else
      \tikz@node@phantomfalse
      \let\tikz@textcolor\tikz@temp
    \fi
}
\usepackage{etoolbox}
\patchcmd\tikz@fig@continue{\tikz@node@transformations}{%
  \iftikz@node@phantom
    \setbox\pgfnodeparttextbox\hbox{}
  \fi\tikz@node@transformations}{}{}
\makeatother

% Recursive style definition idea from
% https://tex.stackexchange.com/questions/55948/choosing-styles-conditionally-in-tikz
\tikzset{ vertexPlain/.code={
    vertexNodeStyle/.style={shape=circle, inner sep=0pt, minimum size=2pt, fill=gray, text=none},
    \ifdefined\vertexLabels
        vertexLabelStyle/.style={gray}
    \else
        vertexLabelStyle/.style={text=none}
    \fi
} }
\tikzset{ vertexNormal/.code={
    vertexNodeStyle/.style={shape=circle, inner sep=0pt, minimum size=4pt, fill=blue, text=none},
    \ifdefined\vertexLabels
        vertexLabelStyle/.style={blue}
    \else
        vertexLabelStyle/.style={text=none}
    \fi
} }
\tikzset{ vertexBall/.code={
    \ifdefined\vertexLabels
        vertexNodeStyle/.style={shape=circle, ball color=orange, text=black, inner sep=2pt, outer sep=0pt, minimum size=3pt},
    \else
        vertexNodeStyle/.style={shape=circle, ball color=orange, text=none, inner sep=2pt, outer sep=0pt, minimum size=3pt},
    \fi
    vertexLabelStyle/.style={text=none}
} }
\tikzset{ vertexDefault/.style={vertexNormal} }
\tikzset{ vertexNode/.style={vertexDefault, vertexNodeStyle} }
\tikzset{ vertexLabel/.style={vertexDefault, vertexLabelStyle} }


%%% Edge styles
\tikzset{ edgePlain/.style={join line=round} }
\tikzset{ edgeDouble/.style={thin, double=gray!90!white, double distance=.3pt, join line=round} }
\tikzset{ edgeDefault/.style={edgePlain} }
\tikzset{ edge/.code={
    \ifdefined\edgeLabels
        \tikzset{fill=blue!20!white, edgeDefault}
    \else
        \tikzset{edgeDefault}
    \fi}
}
\newcommand{\edge}[1]{
    \ifdefined\edgeLabels #1 \fi
}


%%% Face styles

\tikzset{ faceG/.style={face=green!60!white} } % green
\tikzset{ faceO/.style={face=orange!50!yellow!70!white} } % orange

\tikzset{ face/.default={\faceColorY} }
