\makeatletter
\newif\iftikz@node@phantom
\tikzset{
  phantom/.is if=tikz@node@phantom,
  text/.code=%
    \edef\tikz@temp{#1}%
    \ifx\tikz@temp\tikz@nonetext
      \tikz@node@phantomtrue
    \else
      \tikz@node@phantomfalse
      \let\tikz@textcolor\tikz@temp
    \fi
}
\usepackage{etoolbox}
\patchcmd\tikz@fig@continue{\tikz@node@transformations}{%
  \iftikz@node@phantom
    \setbox\pgfnodeparttextbox\hbox{}
  \fi\tikz@node@transformations}{}{}
\makeatother

\tikzset{ vertexPlain/.style={
    vertexNode/.style={shape=circle, inner sep=0pt, minimum size=2pt, fill=gray},
    vertexLabel/.style={gray}
} }
\tikzset{ vertexNormal/.style={
    vertexNode/.style={shape=circle, inner sep=0pt, minimum size=4pt, fill=blue, text=none},
    vertexLabel/.style={blue}
} }
\tikzset{ vertexBall/.style={
    vertexNode/.style={shape=circle, ball color=orange, text=black, inner sep=2pt, outer sep=0pt, minimum size=3pt},
    vertexLabel/.style={text=none}
} }
\tikzset{ vertexDefault/.style={vertexNormal} }


\tikzset{ edgePlain/.style={} }
\tikzset{ edgeDouble/.style={thin, double=gray!90!white, double distance=.3pt} }
\tikzset{ edgeDefault/.style={edgePlain} }
\tikzset{ edge/.code={
    \ifdefined\edgeLabels
        \tikzset{fill=blue!20!white, edgeDefault}
    \else
        \tikzset{edgeDefault}
    \fi}
}

    \ifdefined\edgeLabels #1 \fi
