\newcommand{\drawFourRing}{
    % Use parameter \len, \win
    \coordinate (Z) at (0,0);
    \foreach \i in {0,1,2,3,4}{
        \coordinate (P\i) at (\i*\win:\len);
        \draw (Z) -- (P\i);
    }
    \foreach \i/\j in {0/1,1/2,2/3,3/4}
        \draw (P\i) -- (P\j);
}
