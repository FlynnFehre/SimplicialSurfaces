\documentclass[11pt]{scrartcl}

\usepackage{pgf, tikz}
\usetikzlibrary{arrows, calc}

\newcommand{\colA}{blue}
\newcommand{\colB}{red}
\newcommand{\colC}{green!80!black}


\newcommand{\drawOctaNet}{
    %    Q23 -- P2 -- Q12
    %      \   /|\   /
    %       \ / | \ /
    %        P3 | P1
    %       /|\ | /|\
    %      / | \|/ | \
    %    Q34 |  Z  | Q01
    %      \ | /|\ | /
    %       \|/ | \|/
    %        P4 - P0
    \coordinate (Z) at (0,0);
    \foreach \i in {0,1,2,3,4}
        \coordinate (P\i) at (-30+60*\i:4.8cm);
    \foreach \i/\j in {0/1,1/2,2/3,3/4}
        \coordinate (Q\i\j) at ($(P\i)+(P\j)$);


    % Edges
    \draw[\colA,innerEdge] (Z) -- (P2); % inner
    \draw[\colA,outerEdge] (Q23) -- (P2) -- (Q12); % boundary
    \draw[\colA,outerEdge] (Q34) -- (P4) -- (Z) -- (P0) -- (Q01); % boundary

    \draw[\colB,innerEdge] (P3) -- (Z) -- (P1); % inner
    \draw[\colB,outerEdge] (Q34) -- (P3) -- (Q23); % boundary
    \draw[\colB,outerEdge] (Q12) -- (P1) -- (Q01); % boundary

    \draw[\colC,innerEdge] (P4) -- (P3) -- (P2) -- (P1) -- (P0); % inner

    \draw (P0) -- (P4);
    \draw (Z) -- ($(P0)!0.5!(P4)$);

    % Edge labels
    \def\bd{0.2} % barycentric distance
    \node[edgeNode,\colA,rotate=60] at (barycentric cs:Q34=1,P4=1,P3=\bd) {8};
    \node[edgeNode,\colB,rotate=120] at (barycentric cs:Q34=1,P3=1,P4=\bd) {6};
    \node[edgeNode,\colA,rotate=120] at (barycentric cs:P4=1,Z=1,P3=\bd) {4};
    \node[edgeNode,\colB] at (barycentric cs:P3=1,Q23=1,P2=\bd) {7};
    \node[edgeNode,\colA,rotate=60] at (barycentric cs:P2=1,Q23=1,P3=\bd) {5};
    \node[edgeNode,\colA,rotate=-60] at (barycentric cs:P2=1,Q12=1,P1=\bd) {6};
    \node[edgeNode,\colB] at (barycentric cs:P1=1,Q12=1,P2=\bd) {8};
    \node[edgeNode,\colA,rotate=-120] at (barycentric cs:Z=1,P0=1,P1=\bd) {3};
    \node[edgeNode,\colA,rotate=-60] at (barycentric cs:P0=1,Q01=1,P1=\bd) {7};
    \node[edgeNode,\colB,rotate=-120] at (barycentric cs:P1=1,Q01=1,P0=\bd) {5};
    

    % Face names
    \node[stdNode, rotate=-30] at (barycentric cs:Z=1,P1=1,P2=1) {1};
    \node[stdNode, rotate=-30] at (barycentric cs:P1=1,P2=1,Q12=1) {5};
    \node[stdNode, rotate=30] at (barycentric cs:Z=1,P2=1,P3=1) {2};
    \node[stdNode, rotate=30] at (barycentric cs:P2=1,P3=1,Q23=1) {6};
    \node[stdNode, rotate=90] at (barycentric cs:Z=1,P3=1,P4=1) {3};
    \node[stdNode, rotate=90] at (barycentric cs:P3=1,P4=1,Q34=1) {7};
    \node[stdNode, rotate=-90] at (barycentric cs:Z=1,P0=1,P1=1) {4};
    \node[stdNode, rotate=-90] at (barycentric cs:P0=1,P1=1,Q01=1) {8};

    
    % Draw vertices
    \foreach \p in {Z,P0,P1,P2,P3,P4,Q01,Q12,Q23,Q34}
        \fill[gray] (\p) circle (1pt);

}
