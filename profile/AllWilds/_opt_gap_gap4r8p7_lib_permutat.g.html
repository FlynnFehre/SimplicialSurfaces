16##  <#GAPDoc Label="[1]{permutat}">
17##  Internally, &GAP; stores a permutation as a list of the <M>d</M> images
18##  of the integers <M>1, \ldots, d</M>, where the <Q>internal degree</Q>
19##  <M>d</M> is the largest integer moved by the permutation or bigger.
20##  When a permutation is read in in cycle notation, <M>d</M> is always set
21##  to the largest moved integer, but a bigger <M>d</M> can result from a
22##  multiplication of two permutations, because the product is not shortened
23##  if it fixes&nbsp;<M>d</M>.
24##  The images are stored all as <M>16</M>-bit integers or all as
25##  <M>32</M>-bit integers, depending on whether <M>d \leq 65536</M> or not.
26##  For example, if <M>m\geq 65536</M>, the permutation 
27##  <M>(1, 2, \ldots, m)</M> has internal degree <M>d=m</M> and takes
28##  <M>4m</M> bytes of memory for storage. But --- since the internal degree
29##  is not reduced  --- this
30##  means that the identity permutation <C>()</C> calculated as
31##  <M>(1, 2, \ldots, m) * (1, 2, \ldots, m)^{{-1}}</M> also
32##  takes <M>4m</M> bytes of storage.
33##  It can take even more because the internal list has sometimes room for
34##  more than <M>d</M> images.
35##  <P/>
36##  The operation <Ref Func="RestrictedPerm"/> reduces the storage degree of
37##  its result and therefore can be used to save memory if intermediate
38##  calculations in large degree result in a small degree result.
39##  <P/>
40##  Permutations do not belong to a specific group.
41##  That means that one can work with permutations without defining
42##  a permutation group that contains them.
43##  <P/>
44##  <Example><![CDATA[
45##  gap> (1,2,3);
46##  (1,2,3)
47##  gap> (1,2,3) * (2,3,4);
48##  (1,3)(2,4)
49##  gap> 17^(2,5,17,9,8);
50##  9
51##  gap> OnPoints(17,(2,5,17,9,8));
52##  9
53##  ]]></Example>
54##  <P/>

56##  of a list according to a permutation.

545#F  MappingPermListList( <src>, <dst> ) . . perm. mapping one list to another
546##
547##  <#GAPDoc Label="MappingPermListList">
548##  <ManSection>
549##  <Func Name="MappingPermListList" Arg='src, dst'/>
550##
551##  <Description>
552##  Let <A>src</A> and <A>dst</A> be lists of positive integers of the same
553##  length, such that neither may contain an element twice.
554##  <Ref Func="MappingPermListList"/> returns a permutation <M>\pi</M> such
555##  that <A>src</A><C>[</C><M>i</M><C>]^</C><M>\pi =</M>
556##  <A>dst</A><C>[</C><M>i</M><C>]</C>.
557##  The permutation <M>\pi</M> fixes all points larger than the maximum of
558##  the entries in <A>src</A> and <A>dst</A>.
559##  If there are several such permutations, it is not specified which of them
560##  <Ref Func="MappingPermListList"/> returns.
561##  </Description>
562##  </ManSection>
563##  <#/GAPDoc>
564##
565BIND_GLOBAL( "MappingPermListList", function( src, dst )
566
567    if not IsList(src) or not IsList(dst) or Length(src) <> Length(dst)  then
568       Error("usage: MappingPermListList( <lst1>, <lst2> )");
569    fi;
570
571    if IsEmpty( src )  then
572        return ();
573    fi;
574
575    src := Concatenation( src, Difference( [1..Maximum(src)], src ) );
576    dst := Concatenation( dst, Difference( [1..Maximum(dst)], dst ) );
577    src := PermList(src);
578    if src = fail then return fail; fi;
579    dst := PermList(dst);
580    if dst = fail then return fail; fi;
581
582    return LeftQuotient( src, dst );
583end );
