1007##
1008#F  ClosureMagmaDefault( <M>, <elm> ) . . . . . closure of magma with element
1009##
1010BindGlobal( "ClosureMagmaDefault", function( M, elm )
1011
1012    local   C,          # closure of `M' with `obj', result
1013            gens,       # generators of `M'
1014            gen,        # generator of `M' or `C'
1015            Celements,  # intermediate list of elements
1016            len;        # current number of elements
1017
1018    gens:= GeneratorsOfMagma( M );
1019
1020    # try to avoid adding an element to a magma that already contains it
1021    if   elm in gens
1022      or ( HasAsSSortedList( M ) and elm in AsSSortedList( M ) )
1023    then
1024        return M;
1025    fi;
1026
1027    # make the closure magma
1028    gens:= Concatenation( gens, [ elm ] );
1029    C:= MagmaByGenerators( gens );
1030    UseSubsetRelation( C, M );
1031    
1032    # if the elements of <M> are known then extend this list
1033    # (multiply each element from the left and right with the new
1034    # generator, and then multiply with all elements until the
1035    # list becomes stable)
1036    if HasAsSSortedList( M ) then
1037
1038        Celements := ShallowCopy( AsSSortedList( M ) );
1039        AddSet( Celements, elm );
1040        UniteSet( Celements, Celements * elm );
1041        UniteSet( Celements, elm * Celements );
1042        repeat
1043            len:= Length( Celements );
1044            for gen in Celements do
1045                UniteSet( Celements, Celements * gen );
1046                UniteSet( Celements, gen * Celements );
1047            od;
1048        until len = Length( Celements );
1049
1050        SetAsSSortedList( C, AsSSortedList( Celements ) );
1051        SetIsFinite( C, true );
1052        SetSize( C, Length( Celements ) );
1053
1054    fi;
1055
1056    # return the closure
1057    return C;
1058end );

1062##
1063#M  Enumerator( <M> ) . . . . . . . . . . . .  set of the elements of a magma
1064##
1065BindGlobal( "EnumeratorOfMagma", function( M )
1066
1067    local   gens,       # magma generators of <M>
1068            H,          # submagma of the first generators of <M>
1069            gen;        # generator of <M>
1070
1071    # The following code only does not work infinite magmas.
1072    if HasIsFinite( M ) and not IsFinite( M ) then
1073      TryNextMethod();
1074    fi;
1075
1076    # handle the case of an empty magma
1077    gens:= GeneratorsOfMagma( M );
1078    if IsEmpty( gens ) then
1079      return [];
1080    fi;
1081
1082    # start with the empty magma and its element list
1083    H:= Submagma( M, [] );
1084    SetAsSSortedList( H, Immutable( [ ] ) );
1085
1086    # Add the generators one after the other.
1087    # We use a function that maintains the elements list for the closure.
1088    for gen in gens do
1089      H:= ClosureMagmaDefault( H, gen );
1090    od;
1091
1092    # return the list of elements
1093    Assert( 2, HasAsSSortedList( H ) );
1094    return AsSSortedList( H );
1095end );
